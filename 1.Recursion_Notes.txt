#include <stdio.h>
void display(int n)
{
    if(n>0){
        printf("%d ",n);
        n--;
        display(n);
    }
}
int main() {
    int n=5;
    display(n);
    return 0;
}

#include <stdio.h>
int sum(int n)
{
    if (n==1)    
       return 1;
    else
      return ((2*n-1)+sum(n-1));
}
int main() {
    int n=4;
    printf("%d", sum(n));
    return 0;
}
/*
   sum of all natural number
     1 + 2 + 3 + 4 + 5.........................N   if n=3 sum=6
     1   2   3   4   5........................ n
     sum(n) = n + sum(n-1) n>1
            = 1            n=1
    
    sum of all odd number
    1 + 3 + 5 + 7 + 9..........................N(2n-1)   if n=3 sum=9
    1   2   3   4   5........................  n
     sum(n) = 2n-1 + sum(n-1) n>1
            = 1               n=1  
   
    1 - 3 + 5 - 7................2n-1*(-1)^n-1   if n=4 sum=-4
    1   2   3   4         	 n	
  
    sum(n)  = (2n-1) *(-1)^n-1 + sum(n-1) n>1
            =  1                          n=1 
     
*/

int fibo(int n)
{
    if (n<=1)

      return (fibo(n-1)+ fibo(n-2));
}
