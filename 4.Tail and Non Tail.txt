

/*Tail Recursion (When Recursive call is the last thing done by the funtion. 
  There is no need to keep the record of previous state)*/
#include <stdio.h>
int fun(int n){
  if (n==0)
    return 0;
  else
    printf("%d  ",n);
  return fun(n-1);
}
int main() {
    printf("%d",fun(4));
    return 0;
}

/*Non-Tail Recursion, need to keep the record of previous state)*/
#include <stdio.h>
int fun(int n){
  if (n==1)
    return 1;
  else
    return 1+fun(n-1);
}
int main() {
    printf("%d",fun(3));
    return 0;
}
/*
   fun(3)=1+fun(2)=3
    fun(2)=1+fun(1)=2
      fun(1)=1
*/
