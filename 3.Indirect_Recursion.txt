//Indirect Recursion


#include <stdio.h>


int f1(int n){
  if (n<=1)
    return 1;
  else
    return n*f2(n-1);
}


int f2(int n){
  if (n<=1)
    return 1;
  else
    return n*f1(n-1);
}
int main() {
    printf("%d",f1(4));
    return 0;
}


/*
       f1(4)= 4*f2(3)=24      f2(3)= 3*f1(2)=6
       
       f1(2)= 2*f2(1)=2       f2(1)=1           
*/
